108 private void DisplayStatisticsTable(List<TaskRecord> tasks)
109 {
110 Console.WriteLine("┌─────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────┐");
111 Console.WriteLine("| # | TASK NAME | START DATE | END DATE | TAGGED | STATUS | PROJECT | CLIENT | TIME SPENT |");
112 Console.WriteLine("├─────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────┤");
113
114 for (int i = 0; i < tasks.Count; i++)
115 {
116 var task = tasks[i];
117 var taskcounter = i + 1;
118
119 string taskName = task.Description.PadRight(20);
120 string startDate = task.StartDate.ToString("MMM dd, yyyy hh:mm tt").PadRight(20);
121 string endDate = task.EndDate == DateTime.MinValue ? "N/A".PadRight(20) : task.EndDate.ToString("MMM dd, yyyy hh:mm tt").PadRight(20);
122 string tagged = string.IsNullOrEmpty(task.Tag) ? "".PadRight(12) : task.Tag.PadRight(12);
123 string status = task.Status.PadRight(10);
124 string project = task.Project.PadRight(12); // Display the Project
125 string client = task.Client.PadRight(12); // Display the Client
126
127 // Calculate the time spent if the task is completed
128 string timeSpent = task.EndDate == DateTime.MinValue
129 ? "N/A".PadRight(12)
130 : (task.EndDate - task.StartDate).ToString(@"hh\:mm\:ss").PadRight(12);
131
132 Console.WriteLine($"| {taskcounter.ToString().PadRight(3)} | {taskName} | {startDate} | {endDate} | {tagged} | {status} | {project} | {client} | {timeSpent} |");
133
134 }
135
136 Console.WriteLine("└─────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────────┘");
137
